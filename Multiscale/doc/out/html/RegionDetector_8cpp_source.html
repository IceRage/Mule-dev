203 void RegionDetector::computeAverageClusterednessDegree(vector<Region> &regions) {
204  avgClusterednessDegree = 0;
205 
206  for (Region &region: regions) {
207  Point2f centroid = region.getCentre();
208  double distance = 0;
209 
210  for (Region &otherRegion: regions) {
211  Point2f otherCentroid = otherRegion.getCentre();
212 
213  distance += Geometry2D::distanceBtwPoints(centroid, otherCentroid);
214  }
215 
216  avgClusterednessDegree += (regions.size() > 1) ? (distance / (regions.size() - 1))
217  : 0;
218  }
219 
220  // Take the average of the average distances between clusters
221  avgClusterednessDegree = (regions.size() != 0) ? avgClusterednessDegree / (regions.size())
222  : 0;
223 
224  // Invert the value such that it is between 0 and 1. Since we are working with pixels
225  // the minimum distance between two distinct pixels is 1.
226  avgClusterednessDegree = (avgClusterednessDegree > 1) ? (1 / avgClusterednessDegree)
227  : 1;
228 }
